# 4. converting and multiplying M with 1000000 and k with 1000

def convert_currency_to_float(value_str):
    if value_str[-1] == 'M':
        return float(value_str[0:-1]) * 1000000
    elif value_str[-1] == 'K':
        return float(value_str[0:-1]) * 1000
    else:
        return float(value_str[0:])
